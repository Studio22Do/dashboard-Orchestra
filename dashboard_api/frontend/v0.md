  createdAt: string

export default function CategoryPage() {
  const router = useRouter()
  const params = useParams()
  const { toast } = useToast()
  const categorySlug = params?.slug as string

  // Verificar si la categoría existe
  const categoryData = CATEGORIES[categorySlug as keyof typeof CATEGORIES]

  // Estados
  const [isLoading, setIsLoading] = useState(true)
  const [searchQuery, setSearchQuery] = useState("")
  const [selectedSubcategory, setSelectedSubcategory] = useState("all")
  const [sortBy, setSortBy] = useState("popular")
  const [filterStatus, setFilterStatus] = useState("all")
  const [viewMode, setViewMode] = useState<"grid" | "list">("grid")
  const [selectedApp, setSelectedApp] = useState<App | null>(null)
  const [isModalOpen, setIsModalOpen] = useState(false)
  const [apps, setApps] = useState<App[]>([])

  // Simular carga de datos
  useEffect(() => {
    const timer = setTimeout(() => {
      if (categoryData) {
        setApps(categoryData.apps)
      }
      setIsLoading(false)
    }, 1500)

    return () => clearTimeout(timer)
  }, [categoryData])

  // Obtener subcategorías únicas para esta categoría específica
  const subcategories = useMemo(() => {
    const subs = [...new Set(apps.map((app) => app.subcategory))]
    return ["all", ...(subs.filter(Boolean) as string[])]
  }, [apps])

  // Filtrar y ordenar apps
  const filteredApps = useMemo(() => {
    let result = [...apps]

    // Filtrar por búsqueda
    if (searchQuery) {
      result = result.filter(
        (app) =>
          app.title.toLowerCase().includes(searchQuery.toLowerCase()) ||
          app.description.toLowerCase().includes(searchQuery.toLowerCase()) ||
          app.apiName.toLowerCase().includes(searchQuery.toLowerCase()),
      )
    }

    // Filtrar por subcategoría
    if (selectedSubcategory !== "all") {
      result = result.filter((app) => app.subcategory === selectedSubcategory)
    }

    // Filtrar por estado
    if (filterStatus === "purchased") {
      result = result.filter((app) => app.isPurchased)
    } else if (filterStatus === "not-purchased") {
      result = result.filter((app) => !app.isPurchased)
    }

    // Ordenar
    if (sortBy === "popular") {
      result.sort((a, b) => (a.is_favorite === b.is_favorite ? 0 : a.is_favorite ? -1 : 1))
    } else if (sortBy === "newest") {
      result.sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime())
    } else if (sortBy === "alphabetical") {
      result.sort((a, b) => a.title.localeCompare(b.title))
    }

    return result
  }, [apps, searchQuery, selectedSubcategory, filterStatus, sortBy])

  // Manejar favoritos
  const toggleFavorite = (appId: string, event: React.MouseEvent) => {
    event.stopPropagation()
    setApps((prevApps) => prevApps.map((app) => (app.id === appId ? { ...app, is_favorite: !app.is_favorite } : app)))

    const app = apps.find((app) => app.id === appId)
    toast({
      title: app?.is_favorite ? "Eliminado de favoritos" : "Añadido a favoritos",
      description: `${app?.title} ha sido ${app?.is_favorite ? "eliminado de" : "añadido a"} tus favoritos`,
      duration: 3000,
    })
  }

  // Manejar compra
  const handlePurchase = (appId: string, event: React.MouseEvent) => {
    event.stopPropagation()
    setApps((prevApps) => prevApps.map((app) => (app.id === appId ? { ...app, isPurchased: true } : app)))

    const app = apps.find((app) => app.id === appId)
    toast({
      title: "Compra exitosa",
      description: `Has comprado ${app?.title} correctamente`,
      duration: 3000,
    })
  }

  // Manejar click en app
  const handleAppClick = (app: App) => {
    setSelectedApp(app)
    setIsModalOpen(true)
  }

  // Si la categoría no existe, mostrar mensaje de error
  if (!categoryData && !isLoading) {
    return (
      <div className="min-h-screen bg-[#1a1625] text-white flex flex-col items-center justify-center p-4">
        <div className="rounded-full bg-[#272038] p-4 mb-4">
          <Search className="h-8 w-8 text-[#837cf2]" />
        </div>
        <h2 className="text-xl font-bold mb-2">Categoría no encontrada</h2>
        <p className="text-muted-foreground max-w-md text-center mb-4">
          La categoría que estás buscando no existe o ha sido eliminada.
        </p>
        <Button onClick={() => router.push("/dashboard")}>Volver al dashboard</Button>
      </div>
    )
  }

  return (
    <div className="min-h-screen bg-[#1a1625] text-white">
      {/* Header */}
      <header className="sticky top-0 z-10 bg-[#1a1625] border-b border-[#3a3045] p-4 md:p-6">
        <div className="flex flex-col gap-4">
          <div className="flex items-center justify-between">
            <div className="flex items-center gap-2">
              <Button
                variant="ghost"
                size="icon"
                onClick={() => router.push("/dashboard")}
                aria-label="Volver al dashboard"
              >
                <ArrowLeft className="h-5 w-5" />
              </Button>
              <div className="flex items-center gap-2">
                <div className="flex h-8 w-8 items-center justify-center rounded-md bg-[#837cf2]">
                  {categoryData?.icon || <Code className="h-5 w-5" />}
                </div>
                <h1 className="text-xl font-bold">{categoryData?.name || "Categoría"}</h1>
              </div>
            </div>

            <div className="hidden md:flex items-center gap-2">
              <Button
                variant="outline"
                size="icon"
                onClick={() => setViewMode(viewMode === "grid" ? "list" : "grid")}
                aria-label={viewMode === "grid" ? "Ver como lista" : "Ver como cuadrícula"}
              >
                {viewMode === "grid" ? <List className="h-4 w-4" /> : <Grid className="h-4 w-4" />}
              </Button>
            </div>
          </div>

          <div className="flex flex-col md:flex-row gap-4 items-start md:items-center justify-between">
            <div className="relative w-full md:w-96">
              <Search className="absolute left-3 top-1/2 h-4 w-4 -translate-y-1/2 text-muted-foreground" />
              <Input
                placeholder="Buscar aplicaciones..."
                className="pl-10 bg-[#272038] border-[#3a3045] focus-visible:ring-[#837cf2]"
                value={searchQuery}
                onChange={(e) => setSearchQuery(e.target.value)}
              />
            </div>

            <div className="flex items-center gap-2 w-full md:w-auto">
              <DropdownMenu>
                <DropdownMenuTrigger asChild>
                  <Button variant="outline" className="flex items-center gap-2">
                    <SlidersHorizontal className="h-4 w-4" />
                    <span>Ordenar</span>
                    <ChevronDown className="h-4 w-4" />
                  </Button>
                </DropdownMenuTrigger>
                <DropdownMenuContent align="end" className="bg-[#272038] border-[#3a3045]">
                  <DropdownMenuItem
                    className={sortBy === "popular" ? "bg-[#3a3045]" : ""}
                    onClick={() => setSortBy("popular")}
                  >
                    Popular
                  </DropdownMenuItem>
                  <DropdownMenuItem
                    className={sortBy === "newest" ? "bg-[#3a3045]" : ""}
                    onClick={() => setSortBy("newest")}
                  >
                    Más reciente
                  </DropdownMenuItem>
                  <DropdownMenuItem
                    className={sortBy === "alphabetical" ? "bg-[#3a3045]" : ""}
                    onClick={() => setSortBy("alphabetical")}
                  >
                    Alfabético
                  </DropdownMenuItem>
                </DropdownMenuContent>
              </DropdownMenu>

              <DropdownMenu>
                <DropdownMenuTrigger asChild>
                  <Button variant="outline" className="flex items-center gap-2">
                    <Filter className="h-4 w-4" />
                    <span>Estado</span>
                    <ChevronDown className="h-4 w-4" />
                  </Button>
                </DropdownMenuTrigger>
                <DropdownMenuContent align="end" className="bg-[#272038] border-[#3a3045]">
                  <DropdownMenuItem
                    className={filterStatus === "all" ? "bg-[#3a3045]" : ""}
                    onClick={() => setFilterStatus("all")}
                  >
                    Todas
                  </DropdownMenuItem>
                  <DropdownMenuItem
                    className={filterStatus === "purchased" ? "bg-[#3a3045]" : ""}
                    onClick={() => setFilterStatus("purchased")}
                  >
                    Compradas
                  </DropdownMenuItem>
                  <DropdownMenuItem
                    className={filterStatus === "not-purchased" ? "bg-[#3a3045]" : ""}
                    onClick={() => setFilterStatus("not-purchased")}
                  >
                    No compradas
                  </DropdownMenuItem>
                </DropdownMenuContent>
              </DropdownMenu>

              <Button
                variant="outline"
                size="icon"
                className="md:hidden"
                onClick={() => setViewMode(viewMode === "grid" ? "list" : "grid")}
                aria-label={viewMode === "grid" ? "Ver como lista" : "Ver como cuadrícula"}
              >
                {viewMode === "grid" ? <List className="h-4 w-4" /> : <Grid className="h-4 w-4" />}
              </Button>
            </div>
          </div>
        </div>
      </header>

      {/* Breadcrumbs */}
      <div className="px-4 md:px-6 py-2 text-sm text-gray-400">
        <div className="flex items-center gap-1">
          <span>Dashboard</span>
          <span>/</span>
          <span className="text-white">{categoryData?.name}</span>
        </div>
      </div>

      {/* Subcategorías - Solo mostrar si hay más de una subcategoría */}
      {subcategories.length > 1 && (
        <div className="px-4 md:px-6 py-4 border-b border-[#3a3045] overflow-x-auto">
          <Tabs
            defaultValue="all"
            value={selectedSubcategory}
            onValueChange={setSelectedSubcategory}
            className="w-full"
          >
            <TabsList className="bg-[#272038]">
              {subcategories.map((sub) => (
                <TabsTrigger
                  key={sub}
                  value={sub}
                  className="data-[state=active]:bg-[#837cf2] data-[state=active]:text-white"
                >
                  {sub === "all" ? "Todas" : sub}
                </TabsTrigger>
              ))}
            </TabsList>
          </Tabs>
        </div>
      )}

      {/* Contenido principal */}
      <main className="p-4 md:p-6">
        {isLoading ? (
          // Skeleton loader
          <div
            className={`grid gap-4 ${
              viewMode === "grid" ? "grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4" : "grid-cols-1"
            }`}
          >
            {Array.from({ length: 4 }).map((_, index) => (
              <Card key={index} className="bg-[#272038] border-[#3a3045] overflow-hidden">
                <Skeleton className="h-48 w-full" />
                <CardContent className="p-4">
                  <div className="space-y-3">
                    <Skeleton className="h-5 w-3/4" />
                    <Skeleton className="h-4 w-full" />
                    <Skeleton className="h-4 w-full" />
                    <div className="flex items-center justify-between pt-2">
                      <Skeleton className="h-8 w-20" />
                      <Skeleton className="h-8 w-8 rounded-full" />
                    </div>
                  </div>
                </CardContent>
              </Card>
            ))}
          </div>
        ) : filteredApps.length === 0 ? (
          // Mensaje de no resultados
          <div className="flex flex-col items-center justify-center py-12 text-center">
            <div className="rounded-full bg-[#272038] p-4 mb-4">
              <Search className="h-8 w-8 text-[#837cf2]" />
            </div>
            <h2 className="text-xl font-bold mb-2">No se encontraron resultados</h2>
            <p className="text-muted-foreground max-w-md">
              No hay aplicaciones que coincidan con tu búsqueda. Intenta con otros términos o filtros.
            </p>
            <Button
              variant="outline"
              className="mt-4"
              onClick={() => {
                setSearchQuery("")
                setSelectedSubcategory("all")
                setFilterStatus("all")
                setSortBy("popular")
              }}
            >
              Limpiar filtros
            </Button>
          </div>
        ) : (
          // Grid de apps
          <div
            className={`grid gap-4 ${
              viewMode === "grid" ? "grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4" : "grid-cols-1"
            }`}
          >
            {filteredApps.map((app) => (
              <Card
                key={app.id}
                className={`bg-[#272038] border-[#3a3045] overflow-hidden transition-all duration-300 hover:shadow-lg hover:shadow-[#837cf2]/20 hover:translate-y-[-2px] cursor-pointer ${
                  viewMode === "list" ? "flex flex-row" : ""
                }`}
                onClick={() => handleAppClick(app)}
              >
                <div className={viewMode === "list" ? "w-1/3" : ""}>
                  <Image
                    src={app.imageUrl || "/placeholder.svg"}
                    alt={app.title}
                    width={400}
                    height={200}
                    className={`object-cover w-full ${viewMode === "list" ? "h-full" : "h-48"}`}
                  />
                </div>
                <div className={viewMode === "list" ? "w-2/3" : ""}>
                  <CardContent className="p-4">
                    <div className="flex items-start justify-between">
                      <div>
                        <h3 className="font-bold text-lg mb-1">{app.title}</h3>
                        <p className="text-sm text-gray-300 mb-2">{app.description}</p>
                      </div>
                      <TooltipProvider>
                        <Tooltip>
                          <TooltipTrigger asChild>
                            <Button
                              variant="ghost"
                              size="icon"
                              className="h-8 w-8 text-yellow-400"
                              onClick={(e) => toggleFavorite(app.id, e)}
                              aria-label={app.is_favorite ? "Quitar de favoritos" : "Añadir a favoritos"}
                            >
                              {app.is_favorite ? (
                                <Star className="h-5 w-5 fill-yellow-400" />
                              ) : (
                                <StarOff className="h-5 w-5" />
                              )}
                            </Button>
                          </TooltipTrigger>
                          <TooltipContent>
                            {app.is_favorite ? "Quitar de favoritos" : "Añadir a favoritos"}
                          </TooltipContent>
                        </Tooltip>
                      </TooltipProvider>
                    </div>

                    <div className="flex flex-wrap gap-2 mt-3">
                      {app.subcategory && (
                        <Badge variant="outline" className="bg-[#3a3045] text-xs">
                          {app.subcategory}
                        </Badge>
                      )}
                      <Badge variant="outline" className="bg-[#3a3045] text-xs">
                        API: {app.apiName}
                      </Badge>
                      {app.isPurchased ? (
                        <Badge className="bg-green-600 text-white text-xs">Comprada</Badge>
                      ) : (
                        <Badge className="bg-[#837cf2] text-white text-xs">Disponible</Badge>
                      )}
                    </div>
                  </CardContent>

                  <CardFooter className="p-4 pt-0 flex justify-between">
                    {app.isPurchased ? (
                      <Button className="bg-[#837cf2] hover:bg-[#6c64d3]">Usar ahora</Button>
                    ) : (
                      <Button className="bg-[#837cf2] hover:bg-[#6c64d3]" onClick={(e) => handlePurchase(app.id, e)}>
                        <ShoppingCart className="h-4 w-4 mr-2" />
                        Comprar
                      </Button>
                    )}
                  </CardFooter>
                </div>
              </Card>
            ))}
          </div>
        )}
      </main>

      {/* Modal de detalles */}
      <Dialog open={isModalOpen} onOpenChange={setIsModalOpen}>
        <DialogContent className="bg-[#272038] border-[#3a3045] text-white max-w-3xl">
          {selectedApp && (
            <>
              <DialogHeader>
                <DialogTitle className="text-xl font-bold">{selectedApp.title}</DialogTitle>
                <DialogDescription className="text-gray-300">{selectedApp.description}</DialogDescription>
              </DialogHeader>

              <div className="mt-4">
                <Image
                  src={selectedApp.imageUrl || "/placeholder.svg"}
                  alt={selectedApp.title}
                  width={800}
                  height={400}
                  className="w-full h-64 object-cover rounded-md"
                />

                <div className="mt-4 grid grid-cols-2 gap-4">
                  <div>
                    <h4 className="font-semibold mb-2">Detalles</h4>
                    <ul className="space-y-2">
                      <li className="flex items-center gap-2">
                        <span className="text-gray-400">Categoría:</span>
                        <Badge variant="outline" className="bg-[#3a3045]">
                          {categoryData?.name}
                        </Badge>
                      </li>
                      {selectedApp.subcategory && (
                        <li className="flex items-center gap-2">
                          <span className="text-gray-400">Subcategoría:</span>
                          <Badge variant="outline" className="bg-[#3a3045]">
                            {selectedApp.subcategory}
                          </Badge>
                        </li>
                      )}
                      <li className="flex items-center gap-2">
                        <span className="text-gray-400">API:</span>
                        <Badge variant="outline" className="bg-[#3a3045]">
                          {selectedApp.apiName}
                        </Badge>
                      </li>
                      <li className="flex items-center gap-2">
                        <span className="text-gray-400">Estado:</span>
                        {selectedApp.isPurchased ? (
                          <Badge className="bg-green-600 text-white">Comprada</Badge>
                        ) : (
                          <Badge className="bg-[#837cf2] text-white">Disponible</Badge>
                        )}
                      </li>
                    </ul>
                  </div>

                  <div>
                    <h4 className="font-semibold mb-2">Características</h4>
                    <ul className="space-y-1 text-gray-300">
                      <li className="flex items-center gap-2">
                        <div className="h-1.5 w-1.5 rounded-full bg-[#837cf2]" />
                        <span>Integración con {selectedApp.apiName}</span>
                      </li>
                      <li className="flex items-center gap-2">
                        <div className="h-1.5 w-1.5 rounded-full bg-[#837cf2]" />
                        <span>Actualizaciones gratuitas</span>
                      </li>
                      <li className="flex items-center gap-2">
                        <div className="h-1.5 w-1.5 rounded-full bg-[#837cf2]" />
                        <span>Soporte técnico</span>
                      </li>
                      <li className="flex items-center gap-2">
                        <div className="h-1.5 w-1.5 rounded-full bg-[#837cf2]" />
                        <span>Documentación completa</span>
                      </li>
                    </ul>
                  </div>
                </div>

                <div className="mt-6 flex justify-between">
                  <Button
                    variant="outline"
                    className="border-[#837cf2] text-[#837cf2]"
                    onClick={() => {
                      toggleFavorite(selectedApp.id, { stopPropagation: () => {} } as React.MouseEvent)
                      setIsModalOpen(false)
                    }}
                  >
                    {selectedApp.is_favorite ? (
                      <>
                        <Star className="h-4 w-4 mr-2 fill-yellow-400" />
                        Quitar de favoritos
                      </>
                    ) : (
                      <>
                        <Star className="h-4 w-4 mr-2" />
                        Añadir a favoritos
                      </>
                    )}
                  </Button>

                  {selectedApp.isPurchased ? (
                    <Button className="bg-[#837cf2] hover:bg-[#6c64d3]">Usar ahora</Button>
                  ) : (
                    <Button
                      className="bg-[#837cf2] hover:bg-[#6c64d3]"
                      onClick={() => {
                        handlePurchase(selectedApp.id, { stopPropagation: () => {} } as React.MouseEvent)
                        setIsModalOpen(false)
                      }}
                    >
                      <ShoppingCart className="h-4 w-4 mr-2" />
                      Comprar
                    </Button>
                  )}
                </div>
              </div>
            </>
          )}
        </DialogContent>
      </Dialog>
    </div>
  )
}
